import pandas as pd
import numpy as np

def detect_anomalies(transaction_data):
    # Calculate Z-Score
    threshold = 3
    z_scores = np.abs((transaction_data - transaction_data.mean()) / transaction_data.std())
    
    return z_scores > threshold
